import numpy as np
import pandas as pd
from scipy import stats
import matplotlib.pyplot as plt


def check_variances(seq_data_question, ost_data_question):
    # Levene's Test for equality of variances
    stat, p_value = stats.levene(seq_data_question, ost_data_question)
    print(f"Levene's Test p-value = {p_value:.4f}, stat: {stat} -> equality of variances : {p_value>0.05}")
    return p_value

def check_normality(seq_data_question, ost_data_question):
    # Shapiro-Wilk test for normality
    seq_normality_p = stats.shapiro(seq_data_question).pvalue
    ost_normality_p = stats.shapiro(ost_data_question).pvalue
    print(f"Shapiro-Wilk Test p-value for SEQ = {seq_normality_p:.4f} -> normality : {seq_normality_p>0.05}")
    print(f"Shapiro-Wilk Test p-value for OST = {ost_normality_p:.4f} -> normality : {ost_normality_p>0.05}")
    return seq_normality_p, ost_normality_p

def perform_ttest(seq_data_question, ost_data_question):
    # Perform a standard t-test (Student's t-test)
    t_stat, p_value = stats.ttest_ind(seq_data_question, ost_data_question)
    print(f"t-test: t-statistic = {t_stat:.3f}, p-value = {p_value:.3f}")
    return t_stat, p_value

def perform_mannwhitneyu(seq_data_question, ost_data_question):
    # Perform Mann-Whitney U test (Non-parametric test)
    u_stat, p_value = stats.mannwhitneyu(seq_data_question, ost_data_question)
    print(f"Mann-Whitney U test: U-statistic = {u_stat:.3f}, p-value = {p_value:.3f}")
    return u_stat, p_value

import numpy as np
import seaborn as sns
import matplotlib.pyplot as plt
from scipy.stats import mannwhitneyu

def plot_nasa_tlx_results(seq_data, ost_data, questions, subtitle, limit):
    """
    Plot NASA TLX results as violin plots for SEQ and OST experiments, using Mann-Whitney U test
    and printing the effect size (r).
    Parameters:
        seq_data (pd.DataFrame): DataFrame containing NASA TLX results for SEQ experiment.
        ost_data (pd.DataFrame): DataFrame containing NASA TLX results for OST experiment.
        questions (list): List of NASA TLX question labels.
        subtitle (str): Subtitle for the plot.
        limit (int): Y-axis limit for the plot.
    """
    # Add experiment labels to the input data
    seq_data['Experiment'] = 'SEQ'
    ost_data['Experiment'] = 'OST'

    # Combine the data for plotting
    melted_data = pd.concat([seq_data, ost_data], axis=0)

    # Initialize the plot
    plt.figure(figsize=(15, 8))
    sns.set(style="whitegrid")

    # Loop through each question to create side-by-side violin plots
    for i, question in enumerate(questions):
        question_data = melted_data[['Experiment', question]].dropna()

        # Filter data for the current question
        seq_data_question = seq_data[question]
        ost_data_question = ost_data[question]

        # Perform Mann-Whitney U test
        u_stat, p_value, z_score = perform_mannwhitneyu(seq_data_question, ost_data_question)

        # Calculate the effect size (r)
        effect_size = calculate_cohens_d(seq_data_question, ost_data_question)

        # Print the results
        print_results(p_value, effect_size, question)

        # Create plot (Boxplot + Swarmplot)
        create_boxplot_and_swarmplot(question_data, question, i, len(questions), limit)

    # Adjust layout and show the plot
    plt.tight_layout()
    plt.suptitle(subtitle, y=1.02, fontsize=14)
    plt.savefig(f'/content/{subtitle}.pdf', format='pdf')


def perform_mannwhitneyu(seq_data_question, ost_data_question):
    """
    Perform the Mann-Whitney U test and return U statistic, p-value, and Z-score.
    """
    u_stat, p_value = mannwhitneyu(seq_data_question, ost_data_question, alternative='two-sided')
    print(f"u_stat: {u_stat:.4f}\tp_value: {p_value:.4f}")

    # Z-score calculation for effect size
    n1 = len(seq_data_question)
    n2 = len(ost_data_question)
    z_score = (u_stat - (n1 * n2) / 2) / np.sqrt(n1 * n2 * (n1 + n2 + 1) / 12)

    return u_stat, p_value, z_score

def calculate_cohens_d(seq_data, ost_data):
    """ Calculate Cohen's d for effect size between two independent samples """
    # Calculate the mean and standard deviation for both groups
    mean_1, mean_2 = np.mean(seq_data), np.mean(ost_data)
    std_1, std_2 = np.std(seq_data, ddof=1), np.std(ost_data, ddof=1)

    # Calculate pooled standard deviation
    pooled_std = np.sqrt(((std_1 ** 2) + (std_2 ** 2)) / 2)

    # Cohen's d formula
    d = (mean_1 - mean_2) / pooled_std

    # Return absolute value of Cohen's d (if needed)
    return abs(d)


def print_results(p_value, effect_size, question):
    """
    Print the statistical test results and effect size.
    """
    if p_value < 0.05:
        print(f"Significant difference found for {question} with cohens d effect size {effect_size:.4f}!\n")
    else:
        print(f"No significant difference for {question}. cohens d  Effect size: {effect_size:.4f}\n")


def create_boxplot_and_swarmplot(question_data, question, i, num_questions, limit):
    """
    Create a boxplot for the given NASA TLX question and its data, overlaid with a swarm plot.
    Parameters:
        question_data (pd.DataFrame): The melted data frame for the question.
        question (str): The NASA TLX question label.
        i (int): The index of the current question in the plot grid.
        num_questions (int): Total number of questions being plotted.
        limit (int): Limit for the y-axis (e.g., 100).
    """
    plt.subplot(1, num_questions, i + 1)  # Create a subplot for each question

    # Create the box plot for the current question
    sns.boxplot(
        x='Experiment',
        y=question,
        data=question_data,
        showfliers=False,
        linewidth=1,
        dodge=False,
        hue='Experiment',
        palette=condition_palette,
        order=condition_order,
        medianprops={'color': '#6E6E6E', 'linewidth': 2}
    )

    # Overlay the swarm plot for the current question (show individual points)
    sns.swarmplot(
        x='Experiment',
        y=question,
        data=question_data,
        color='k',
        size=3,
        alpha=0.6,
        dodge=False,
    )

    # Adjust title formatting if the question is long
    title = question if len(question) < 20 else " ".join(map(str, question.split()[0:4])) + "\n" + " ".join(map(str, question.split()[4:]))

    # Customize the plot for the current question
    plt.title(title, fontsize=12)
    plt.xlabel('Stimulus')  # Label for x-axis
    if i == 0:
        plt.ylabel('Score')  # Y-axis label for score
    else:
        plt.ylabel('')
    plt.ylim(0, limit)  # Limit the y-axis to 0-100
    plt.xticks(fontsize=8)

from scipy.stats import chisquare

# Observed frequencies for categories A and B
observed = [5, 3]

# Expected frequencies assuming a 50/50 distribution

# Perform Chi-Square Goodness of Fit Test

# Output the results
